\section{Empirical Strategy}
\subsection{Stock Model (Agrawal et. al 2025)}
\textbf{Goal}: Do states with lower taxes have a larger population of pilots?

	\begin{equation*}
	\ln \biggr ( \frac{N_{s, t}}{N_{s', t}} \biggr)=  \beta \ln \biggr(\frac{1 - \tau_{s, t}}{1 - \tau_{s', t}} \biggr) + \gamma_s + \gamma_t +  u_{s, t}
\end{equation*}
where
\medskip
\begin{itemize}
	\small
	\item $s$ indexes state, $s'$ indexes the reference state, and $t$ indexes year.
	\medskip
	\item LHS:  population of pilots in state $s$ relative to the reference state
	\medskip
	\item RHS: net-of-income-tax rate in state $s$ relative to the reference state
	\medskip
	\item $ \gamma_s$ and $\gamma_t$:  state and year fixed effects
\end{itemize}

	\subsection{Migration Log Odds Ratio (Moretti and Wilson 2017)}
	\textbf{Goal}: Do destination-origin state tax differentials affect the probability of moving from the destination state to the origin state? \\
	
	Regression:
	\begin{align*} 
		\underbrace{\log \Big(\frac{P_{odt}}{P_{oot}}\Big)}_{\substack{\text{migration}\\\text{log odds ratio}}}
		= \beta_0 + \beta_1 \underbrace{[\log(1 - \tau_{dt}) - \log(1 - \tau_{ot})]}_{\substack{\text{log dest-origin}\\\text{net-of-tax diff.}}}
		+ \underbrace{\gamma_t}_{\substack{\text{year}\\\text{FE}}}
		+ \underbrace{\gamma_d}_{\substack{\text{destination}\\\text{state FE}}}
		+ \underbrace{\gamma_o}_{\substack{\text{origin}\\\text{state FE}}}
		+ \underbrace{\gamma_{od}}_{\substack{\text{state-pair}\\\text{FE}}}
		+ u_{odt}
	\end{align*}
	where $o$ indexes the origin state, $d$ indexes the destination state, and $t$ indexes year.
	\begin{itemize}
		\item LHS: migration log odds ratio, between the number of pilots who moved from the origin state $o$ to the destination state $d$ and the number of pilots who stayed in the origin state $o$ in year $t$.
		\item RHS: log net-of-average-personal-income tax rate difference between the destination state and the origin state.
	\end{itemize}

\subsection{Individual-Level Linear Probability Model}

\textbf{Goal:} Are pilots less likely to move out of the origin state if the net-of-tax rate is high? \\

Regression:
\begin{equation*}
	\text{Moved}_{i,o,t} =  \beta (1 - \tau_{o, t}) + \gamma_o + \gamma_t +  \gamma_i  +u_{i, o, t}
\end{equation*}
\begin{itemize}
	\item LHS: 1 if moved, 0 if not.
	\medskip
	\item RHS: net-of-ATR in origin state 
	\medskip
	\item $ \gamma_o$, $\gamma_t$, and $\gamma_i$:  origin state, year, and pilot fixed effects
\end{itemize}
